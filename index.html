  <script>

    function openAll(urls) {

      const status = document.getElementById('status');

      status.textContent = 'Opening tabs...';

      let opened = 0;

      urls.forEach((u, i) => {

        setTimeout(() => {

          const w = window.open(u, '_blank', 'noopener');

          if (w) opened++;

          if (i === urls.length - 1) {

            status.textContent = opened === urls.length

              ? 'Opened all tabs successfully.'

              : `Opened ${opened}/${urls.length}. If some didn’t open, allow pop‑ups for this site and try again.`;

          }

        }, i * 220);

      });

    }
